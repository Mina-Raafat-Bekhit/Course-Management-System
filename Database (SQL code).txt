CREATE TABLE  "COURSES" 
   (	"ID" NUMBER(30,0) NOT NULL ENABLE, 
	"NAME" VARCHAR2(30), 
	"HOURS" NUMBER(30,0), 
	"PROF_NAME" VARCHAR2(30), 
	"NO_OF_STUDENT" NUMBER(38,0), 
	 CONSTRAINT "COURSES_PK" PRIMARY KEY ("ID") ENABLE
   ) ;

CREATE OR REPLACE TRIGGER  "BI_COURSES" 
  before insert on "COURSES"               
  for each row  
begin   
  if :NEW."ID" is null then 
    select "COURSES_SEQ".nextval into :NEW."ID" from dual; 
  end if; 
end;
